Def CreatureIs0(X)
  Not CreatureFlag X
End

Def CreatureIs1(X)
  CreatureFlag X
End

Def CreatureWrite0(X)
  SetCreatureFlag X false
End

Def CreatureWrite1(X)
  SetCreatureFlag X true
End

Def CreatureGteBit(A, B)
  CreatureIs1(A)
End

Def CreatureGtBit(A, B)
  And { CreatureIs1(A) CreatureIs0(B) }
End

Def CreatureEqBit(A, B)
  Or { And { CreatureIs0(A) CreatureIs0(B) } And { CreatureIs1(A) CreatureIs1(B) } }
End

Def CreatureLteBit(A, B)
  CreatureIs0(A)
End

Def CreatureLtBit(A, B)
  And { CreatureIs0(A) CreatureIs1(B) }
End

Def CreatureCopy(A, B)
  if(CreatureIs1(A)){
    CreatureWrite1(B)
  }
  else(){
    CreatureWrite0(B)
  }
  endif()
End

Def CreatureCopyToGlobal(A, B)
  if(CreatureIs1(A)){
    Write1(B)
  }
  else(){
    Write0(B)
  }
  endif()
End

Def CreatureIncBits(A, B, Cont)
  if(CreatureIs0(A)) {
    CreatureWrite1(A)
  }
  else() {
    CreatureWrite0(A)
    if(CreatureIs0(B)) {
      CreatureWrite1(B)
    }
    else() {
      CreatureWrite0(B)
      Cont
      CreatureWrite0(B) CreatureWrite0(A)
    }
    endif()
  }
  endif()
End

Def CreatureDecBits(A, B, Cont)
  if(CreatureIs1(A)) {
    CreatureWrite0(A)
  }
  else() {
    CreatureWrite1(A)
    if(CreatureIs1(B)) {
      CreatureWrite0(B)
    }
    else() {
      CreatureWrite1(B)
      Cont
    }
    endif()
  }
  endif()
End
